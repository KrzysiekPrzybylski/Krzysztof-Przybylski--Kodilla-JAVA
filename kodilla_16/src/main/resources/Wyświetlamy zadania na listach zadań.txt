select I.ID, I.ISSUESLIST_ID, I.SUMMARY, I.DESCRIPTION, I.USER_ID_ASSIGNEDTO, L.NAME
from ISSUES I
join ISSUESLISTS L on L.ID = I.ISSUESLIST_ID;
commit;

from USERS U
join ISSUES I on U.ID = I.ISSUESLIST_ID;
COMMIT;
